import pandas as p
# creating dataframe with key value pair key --> column name and value --> list
df = p.DataFrame(
    {
      'Cricket': [4,6,5,6],
      'Football': [5,4,5,8],
      'Tennis':[4,6,2,7],
      'Hockey': [4,6,8,7]  
    }
)
def cal(a,b,c,d):
  point = 0
  if a>5: 
    point += 1
  if b>5:
    point += 1
  if c>5 or d>5:
    point += 1
  return point
df['Points'] = df.apply(lambda x:cal(x['Cricket'],x['Football'],x['Tennis'],x['Hockey']),axis=1)
